```python
import requests
import os
from dotenv import load_dotenv

# Load environment variables
load_dotenv()
serper_api_key = os.getenv("SERPER_API_KEY")

# Try multiple search queries to get more restaurants
queries = [
    "restaurants Mundelein Illinois",
    "best restaurants Mundelein IL",
    "top rated restaurants Mundelein"
]

url = "https://google.serper.dev/search"
headers = {
    "X-API-KEY": serper_api_key,
    "Content-Type": "application/json"
}

all_restaurants = []

for query in queries:
    payload = {
        "q": query,
        "hl": "en",
        "gl": "us",
        "num": 20
    }
    response = requests.post(url, headers=headers, json=payload)
    
    if response.status_code == 200:
        data = response.json()
        restaurants = data.get("organic", [])
        all_restaurants.extend(restaurants)

# Extract relevant data and sort by rating
restaurant_data = []
seen_names = set()

for restaurant in all_restaurants:
    title = restaurant.get("title", "")
    snippet = restaurant.get("snippet", "")
    rating = restaurant.get("rating")
    link = restaurant.get("link", "")
    
    # Extract restaurant name - use title as primary source
    name = title
    
    # Remove common website suffixes
    suffixes = [" - Yelp", " - Tripadvisor", " - Google", " | OpenTable", " - Menu", " - Reviews", 
                " - Mundelein", " - Mundelein, IL", " - Mundelein IL", " Restaurant"]
    for suffix in suffixes:
        if suffix in name:
            name = name.split(suffix)[0].strip()
    
    # Skip obvious list/guide pages
    list_indicators = ["TOP 10", "BEST 10", "10 BEST", "RESTAURANTS NEAR", "DINING NEAR", "HIDDEN GEMS", "RECOS"]
    if any(indicator in title.upper() for indicator in list_indicators):
        continue
    
    # Skip Reddit posts and non-restaurant links
    if "reddit.com" in link or "r/" in title:
        continue
    
    # Skip if name is too short or generic
    if not name or len(name) < 3:
        continue
    
    # Avoid duplicates
    name_lower = name.lower()
    if name_lower in seen_names:
        continue
    seen_names.add(name_lower)
    
    # Use rating if available
    restaurant_rating = rating if rating else 0
    restaurant_data.append({"name": name, "rating": restaurant_rating})

# Sort restaurants by rating (highest first) and get top 50
top_restaurants = sorted(restaurant_data, key=lambda x: x['rating'], reverse=True)[:50]

# Print the top 50 restaurants
print("Top Restaurants in Mundelein, IL:")
print("=" * 60)
for i, restaurant in enumerate(top_restaurants, 1):
    if restaurant['rating'] > 0:
        print(f"{i}. {restaurant['name']} - Rating: {restaurant['rating']}")
    else:
        print(f"{i}. {restaurant['name']}")

print(f"\nTotal restaurants found: {len(top_restaurants)}")
```
